399  typedef struct _tVZFilter
400  {
401  tMempool mempool;
402 
403  VZFilterType type;
404  // state:
405  float s1, s2;
406 
407  // filter coefficients:
408  float g; // embedded integrator gain
409  float R2; // twice the damping coefficient (R2 == 2*R == 1/Q)
410  float h; // factor for feedback (== 1/(1+2*R*g+g*g))
411  float cL, cB, cH; // coefficients for low-, band-, and highpass signals
412 
413  // parameters:
414  float fc; // characteristic frequency
415  float G; // gain
416  float invG; //1/gain
417  float B; // bandwidth (in octaves)
418  float m; // morph parameter (0...1)
419 
420  float sr; //local sampling rate of filter (may be different from leaf sr if oversampled)
421  float inv_sr;
422  } _tVZFilter;
