973

975  float g; // embedded integrator gain

977  float h; // factor for feedback (== 1/(1+2*R*g+g*g))

979 
980  // parameters:

982  float G; // gain
983  float invG; //1/gain
984  float B; // bandwidth (in octaves)
985  float m; // morph parameter (0...1)
