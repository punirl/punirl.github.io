59  /* Feedback leveller */
60  // An auto VCA that you put into a feedback circuit to make it stay at the same level.
61  // It can enforce level bidirectionally (amplifying and attenuating as needed) or
62  // just attenutating. The former option allows for infinite sustain strings, for example, while
63  // The latter option allows for decaying strings, which can never exceed
64  // a specific level.
65 
66  typedef struct _tFeedbackLeveler
67  {
68  tMempool mempool;
69  float targetLevel; // target power level
70  float strength; // how strongly level difference affects the VCA
71  int mode; // 0 for upwards limiting only, 1 for biderctional limiting
72  float curr;
73  tPowerFollower pwrFlw; // internal power follower needed for level tracking
74 
75  } _tFeedbackLeveler;
